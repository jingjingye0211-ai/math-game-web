packages = ["opencv-python", "numpy"] import cv2 import numpy as np import random import time from pyodide import create_proxy from js import document, navigator, console from pyodide.ffi import to_js # ===================== 全局变量（修复作用域问题） ===================== current_difficulty = 20 question = "" answer = 0 correct_num = 0 wrong_num = 0 star_count = 0 star_threshold = 5 last_correct_time = 0 cooldown = 1.5 # 答题冷却时间 opencv_loaded = False cap = None video_elem = document.getElementById("video") question_elem = document.getElementById("question") finger_count_elem = document.getElementById("finger-count") feedback_elem = document.getElementById("feedback") stats_elem = document.getElementById("stats") # ===================== 1. OpenCV加载完成回调 ===================== def opencvReady(): global opencv_loaded opencv_loaded = True console.log("OpenCV loaded successfully!") init_game() # ===================== 2. 生成题目（保留原逻辑：排除整十数） ===================== def generate_question(difficulty=20): while True: a = random.randint(0, difficulty) b = random.randint(0, difficulty) if a < b: a, b = b, a op = random.choice(['+', '-']) ans = a + b if op == '+' else a - b # 排除整十数，且答案非负、不超过难度 if ans % 10 != 0 and ans >= 0 and ans <= difficulty: break return f"{a} {op} {b} = ?", ans # ===================== 3. 优化版手势计数（OpenCV核心，适配网页） ===================== def count_fingers(frame): if not opencv_loaded: return 0 # 1. 转为灰度图 + 降噪 gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY) blur = cv2.GaussianBlur(gray, (7, 7), 0) # 2. 肤色检测（提升抗干扰性，适配手部识别） hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV) lower_skin = np.array([0, 20, 70], dtype=np.uint8) upper_skin = np.array([20, 255, 255], dtype=np.uint8) mask = cv2.inRange(hsv, lower_skin, upper_skin) # 3. 结合灰度图与肤色掩码，分割手部区域 masked_gray = cv2.bitwise_and(gray, gray, mask=mask) # 4. 二值化（自适应阈值，适配不同光线） thresh = cv2.adaptiveThreshold(masked_gray, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C, cv2.THRESH_BINARY_INV, 15, 2) # 5. 轮廓检测（只找最大的轮廓，即手部） contours, _ = cv2.findContours(thresh.copy(), cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE) if not contours: return 0 max_contour = max(contours, key=cv2.contourArea) # 过滤过小的轮廓（排除背景噪点） if cv2.contourArea(max_contour) < 5000: return 0 # 6. 凸包检测（核心：识别手指尖） hull = cv2.convexHull(max_contour, returnPoints=False) defects = cv2.convexityDefects(max_contour, hull) finger_count = 0 # 7. 遍历凸包缺陷，计数手指 if defects is not None: for i in range(defects.shape[0]): s, e, f, d = defects[i, 0] start = tuple(max_contour[s][0]) end = tuple(max_contour[e][0]) far = tuple(max_contour[f][0]) # 计算角度（过滤误判） a = np.sqrt((end[0] - start[0])**2 + (end[1] - start[1])**2) b = np.sqrt((far[0] - start[0])**2 + (far[1] - start[1])**2) c = np.sqrt((end[0] - far[0])**2 + (end[1] - far[1])**2) angle = np.arccos((b**2 + c**2 - a**2) / (2 * b * c)) * 180 / np.pi # 角度小于90度，判定为手指 if angle < 90 and d > 1000: finger_count += 1 # 手指数量 = 缺陷数 + 1（拇指），限制最大为10 return min(finger_count + 1, 10) # ===================== 4. 答题判断（修复冷却逻辑） ===================== def check_answer(user_input): global correct_num, wrong_num, star_count, last_correct_time, question, answer current_time = time.time() # 冷却期内不处理 if current_time - last_correct_time < cooldown: return # 答题判断 if user_input == answer: feedback_elem.innerText = "CORRECT! ✅" feedback_elem.style.color = "#28a745" correct_num += 1 last_correct_time = current_time # 星星奖励 if correct_num % star_threshold == 0: star_count += 1 feedback_elem.innerText = f"CORRECT! ⭐ Stars: {star_count}" # 生成新题 question, answer = generate_question(current_difficulty) question_elem.innerText = question else: feedback_elem.innerText = f"WRONG! Answer is {answer}" feedback_elem.style.color = "#dc3545" wrong_num += 1 # 更新统计 stats_elem.innerText = f"Correct: {correct_num} | Wrong: {wrong_num} | Stars: {star_count}" # ===================== 5. 难度切换（修复按钮事件） ===================== def set_difficulty(diff): global current_difficulty, question, answer current_difficulty = diff question, answer = generate_question(diff) question_elem.innerText = question feedback_elem.innerText = f"Difficulty set to {diff}!" feedback_elem.style.color = "#6a0dad" # ===================== 6. 视频实时处理（修复渲染逻辑） ===================== def process_frame(): if not opencv_loaded or cap is None: # 循环调用自身 setTimeout(create_proxy(process_frame), 30) return # 读取摄像头帧 ret, frame = cap.read() if not ret: setTimeout(create_proxy(process_frame), 30) return # 镜像翻转（符合人眼视角） frame = cv2.flip(frame, 1) # 手势计数 finger_count = count_fingers(frame) # 更新界面显示 finger_count_elem.innerText = f"Fingers: {finger_count}" # 稳定计数后答题（避免手抖误判：计数>0时提交） if finger_count > 0: check_answer(finger_count) # 循环调用，保持实时更新 setTimeout(create_proxy(process_frame), 30) # ===================== 7. 游戏初始化（修复摄像头调用） ===================== def init_game(): global question, answer, cap # 1. 生成初始题目 question, answer = generate_question(current_difficulty) question_elem.innerText = question # 2. 绑定难度按钮事件（修复create_proxy调用） document.getElementById("btn10").addEventListener("click", create_proxy(lambda e: set_difficulty(10))) document.getElementById("btn20").addEventListener("click", create_proxy(lambda e: set_difficulty(20))) document.getElementById("btn100").addEventListener("click", create_proxy(lambda e: set_difficulty(100))) #
